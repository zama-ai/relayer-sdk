export function sleep(ms: number): Promise<void> {
  return new Promise((resolve) => setTimeout(resolve, ms));

async function getDirname(): Promise<string> {
  const { fileURLToPath } = await import("node:url");
  const { dirname } = await import("node:path");
  return dirname(fileURLToPath(import.meta.url));
}

async function readWasmBytes() {
  const dir = await getDirname();
  const wasmBytes = await readFile(
    resolve(dir, "./wasm/tfhe/tfhe_bg.v1.5.3.wasm"),
  );
  return wasmBytes;
}

function preloadTfheBase64FetchBlobThen(numThreads: number) {
  return import("./wasm/tfhe/tfhe_bg.v1.5.3.wasm.base64.js")
    .then(({ tfheWasmBase64 }) =>
      fetch(`data:application/octet-stream;base64,${tfheWasmBase64}`),
    )
    .then((res) => res.arrayBuffer())
    .then((buf) => new Uint8Array(buf))
    .then((wasmBytes) => initTFHE({ module_or_path: wasmBytes }))
    .then(() => {
      init_panic_hook();
      return initThreadPool(numThreads);
    });
}

function compileWasmV2(): Promise<WebAssembly.Module> {
  return new Promise((resolve, reject) => {
    try {
      const worker = new Worker(
        new URL("./wasm/tfhe/compile-worker.js", import.meta.url),
      );
      worker.on("message", (data) => resolve(data));
      worker.on("error", (e) => reject(e));
    } catch (e) {
      console.log(e);
    }
  });
}

async function compileWasmV1(): Promise<WebAssembly.Module> {
  // `data` is the input argument
  const code = `const res = await fetch("data:application/octet-stream;base64," + data);
    const bytes = new Uint8Array(await res.arrayBuffer());
    return await WebAssembly.compile(bytes);`;

  const { tfheWasmBase64 } = await import(
    "./wasm/tfhe/tfhe_bg.v1.5.3.wasm.base64.js"
  );

  const module = runCodeInIsomorphicWorker<WebAssembly.Module>(
    code,
    tfheWasmBase64, // === data
  );

  return module;
}

async function compileWasmV3(): Promise<WebAssembly.Module> {
  const wasmUrl = new URL("./wasm/tfhe/tfhe_bg.v1.5.3.wasm", import.meta.url);

  const isbrowser = isBrowserLike();

  let bytes: BufferSource;

  if (!isbrowser && wasmUrl.protocol === "file:") {
    // Node + file url
    const { readFile } = await import("node:fs/promises");
    const { fileURLToPath } = await import("node:url");

    bytes = await readFile(fileURLToPath(wasmUrl));
  } else {
    // fetch wasm
    const res = await fetch(wasmUrl);

    if (isbrowser) {
      // use compile streaming
      if (typeof WebAssembly.compileStreaming === "function") {
        return WebAssembly.compileStreaming(res);
      }
    }

    bytes = await res.arrayBuffer();
  }

  return WebAssembly.compile(bytes);
}
